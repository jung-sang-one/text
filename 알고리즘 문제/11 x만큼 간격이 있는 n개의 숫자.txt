function solution(x, n) {   //x=x+x의 처음값을 n번 반복하여 n만큼 배열하라
    var answer = [];
    for (let i = 1; i<=n; i++){//i을 선언후 1을 할당하여 우리가 n개만큼 출력을 할거니
                               //n이하로 반복하라는 뜻이며 이때 한번 반복한뒤 i는 1씩 증가
        answer.push(x*i);// 앤써의 배열에 i의 배열수만큼 x를 곱해준다
                         // 앤써의 배열은 한개로 구성되어있다고 처음 선언햇기에
                         // 앤써의 값은 앤써의 배열값 1* x의값 2 를 i의 배열 위치마다의값을 
                         // 곱해준다
    }
    return answer;
}
